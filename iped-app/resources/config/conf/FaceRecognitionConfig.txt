# Face recognition feature
# Dependencies must be installed in the Python used by the subprocess (see pythonPath below).
# For 'auraface' (default — fully open-source, MIT + Apache 2.0):
#   pip install numpy onnxruntime-gpu opencv-python pillow
#   (onnxruntime-gpu includes CPU fallback; use plain 'onnxruntime' for CPU-only machines)
# For 'buffalo_l'/'buffalo_s' (non-commercial InsightFace models):
#   pip install "numpy>=1.26,<2" insightface onnxruntime pillow
# See https://github.com/sepinf-inc/IPED/wiki/User-Manual#facerecognition
# If enabled, you can search for faces from the analysis interface, check the options menu.

# Number of external face recognition processes. Default is min(4, numThreads/2).
# Each process loads the model (~300MB), so more processes = more memory.
# GPU is auto-detected by onnxruntime; multiple processes still help with I/O overlap.
# Set to 1 if GPU memory is limited.
#numFaceRecognitionProcesses = 4

# Max width or height used to resize images before face detection. This dramatically reduces
# the time of face detection on high resolution images, although small faces are lost.
maxResolution = 1024

# Face recognition model. Auto-downloaded on first use.
# Options:
#   'auraface'  - RetinaFace-R50 detection (MIT) + AuraFace-v1 recognition (Apache 2.0) (~365MB) — default
#   'buffalo_l' - InsightFace full pipeline (non-commercial license, ~300MB) — more accurate
#   'buffalo_s' - InsightFace full pipeline (non-commercial license, smaller/faster)
# 'auraface' is fully open-source (MIT + Apache 2.0), no insightface package needed.
# 'buffalo_l'/'buffalo_s' require the insightface package (non-commercial license).
insightFaceModel = auraface

# Detection input size. Larger values detect smaller faces but are slower.
detSize = 640

# Minimum detection confidence score (0.0 to 1.0). Faces below this threshold are discarded.
minDetScore = 0.5

# Minimum size (in pixels) to apply face recognition to an image.
# Images with either dimension (height or width) smaller than this threshold are ignored by this task.
minSize = 48
