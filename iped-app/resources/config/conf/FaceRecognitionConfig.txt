# Face recognition feature based on InsightFace (ArcFace + RetinaFace)
# Dependencies must be installed in the Python used by the subprocess (see pythonPath below).
# CPU-only (simplest): install into IPED's bundled Python:
#   {iped.root}/python/python.exe -m pip install "numpy>=1.26,<2" insightface onnxruntime pillow
# GPU (faster): create a separate Python env with CUDA/cuDNN and point pythonPath to it:
#   pip install "numpy>=1.26,<2" insightface onnxruntime-gpu pillow
# Note: numpy must be <2 because insightface C extensions require the NumPy 1.x ABI.
# See https://github.com/sepinf-inc/IPED/wiki/User-Manual#facerecognition
# If enabled, you can search for faces from the analysis interface, check the options menu.

# Number of external face recognition processes. Default is 1.
# Increase only for CPU-only setups where you want parallelism at the cost of loading the model multiple times.
# When using GPU, keep at 1 — multiple processes compete for the same device.
#numFaceRecognitionProcesses = 1

# Max width or height used to resize images before face detection. This dramatically reduces
# the time of face detection on high resolution images, although small faces are lost.
maxResolution = 1024

# InsightFace model name. Options: 'buffalo_l' (more accurate) or 'buffalo_s' (faster, smaller).
# Models are auto-downloaded on first use (~300MB for buffalo_l).
insightFaceModel = buffalo_l

# Detection input size. Larger values detect smaller faces but are slower.
detSize = 640

# Minimum detection confidence score (0.0 to 1.0). Faces below this threshold are discarded.
minDetScore = 0.5

# Minimum size (in pixels) to apply face recognition to an image.
# Images with either dimension (height or width) smaller than this threshold are ignored by this task.
minSize = 48
