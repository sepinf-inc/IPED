# Detects Child Sexual Abuse Material (CSAM) using a TensorFlow or PyTorch trained AI model.
# Must have either tensorflow, onnxruntime, or pytorch with timm installed, see online documentation.
# To enable the module, set "enableCSAMDetector=true" property in IPEDConfig.txt file
#
# Availabel models, with stats for csam class in test dataset:
#   TENSORFLOW MODELS (for GPU) - requires TensorFlow 2.20
#    	- tensorflow_B0_v3_1.keras    - EfficientNetV2-B0 using tensorflow, image size 224x224 (precision 0.91, recall 0.96)
#
#	TFLITE MODELS (for CPU - requires tensorflow installation)
#		- tflite_B0_v3_1.tflite 	- Fastest model - converted from tensorflow B0 to tflite with 32-bit floating-point (FP32)  (precision 0.89, recall 0.95)
#
#   PYTORCH MODELS (for GPU)
#   	- pytorch_B0_v3_1.pth         - based on efficientnetv2_b0.in1k, image size 224x224 (precision 0.90, recall 0.93)
#    	- pytorch_S_v3_1.pth      	- based on tf_efficientnetv2_s.in21k_ft_in1k, image size 384x384 (recommended the use of GPU)  (precision 0.96, recall 0.97)
#
#	ONNX MODELS (for CPU, requires only onnxruntime installation  - less then 20MB)
#		- onnx_B0_tensorflow_v3_1.onnx		- converted from tensorflow B0 to ONNX FP32 (precision 0.89, recall 0.95)
#
CSAMModelFile = onnx_B0_tensorflow_v3_1.onnx

# Batch size to be used. When using the large models this value should  be reduced to avoid GPU memory exaustion
# For example, recommended size for 12 GB GPU on large models is 16
# Ignored when using TFLITE or ONNX models
CSAMBatchSize = 64

# Minimum image size to be processed, in bytes (use 0 to disable)
CSAMMinimumImageSize = 2048

# Minimum file dimension, in pixels (use 0 to disable)
CSAMSkipDimension = 10

# Skip files with hits on IPED HashDB database (known and csam)
CSAMSkipHashDBFiles = true
