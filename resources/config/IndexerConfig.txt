########################################################################
# Configurações do Processamento
########################################################################

# OS PARSERS PODEM SER HABILITADOS/DESABILITADOS EM conf/ParserConfig.xml

# Diretório temporário de indexação: "default" utiliza o diretório temporário padrão do sistema.
# Configurar indexTemp em outro disco pode aumentar consideravelmente o desempenho, principalmente em um SSD.
indexTemp = default

# Número de threads de processamento: "default" utiliza o número de processadores.
# Caso haja problemas de consumo de memória, configurar um valor pequeno (no mínimo 1) e/ou aumentar memória heap do java (-Xms)
numThreads = default

# Força merge do índice para um único segmento, diminuindo o índice e otimizando a busca a partir de mídias ópticas.
# É muito custoso, sendo desnecessário caso o índice seja acessado a partir de um HD.
forceMerge = false

# Tempo de espera por progresso no parsing dos arquivos. São adicionados 1s por MB.
# Após este limite, são indexadas strings brutas do arquivo.
timeOut = 180

# Parser para arquivos não suportados pelo Tika, como binários, desconhecidos, pagefile, drivefreespace, etc
# Deixe em branco (nao comentado) caso não deseje indexar tais arquivios.
fallBackParser = dpf.sp.gpinf.indexer.parsers.RawStringParser

# Parser para arquivos corrompidos que causem erro de parsing pelo Tika
# Deixe em branco caso não deseje indexar tais arquivos
errorParser = dpf.sp.gpinf.indexer.parsers.RawStringParser

# Tamanho mínimo de strings brutas extraídas pelo RawStringParser
minRawStringSize = 4

# Copia LibreOffice.zip (100MB) para a pasta de saída da indexação, permitindo visualizar mais de 20 formatos diferentes.
embutirLibreOffice = true

# Categoria para arquivos exportados para o relatório, porém não pertencentes a nenhum bookmark
# Ex: Anexos de emails
defaultCategory = Outros Itens Exportados

# Habilita cálculo de hashes dos arquivos
# Valores possíveis: md5, sha-1, sha-256 e sha-512
hash = md5

# Diretório contendo a base de hashes do LED. Caso configurado, os arquivos com hashes
# existentes na base são adicionados a uma categoria "Alerta de hash"
#ledWkffPath = D:/LED/V1.17.00/pedo/wkff

# Caminho para base de hashes (KFF) em formato indexado. Comente caso não queira confrontar hashes na base.
# A base pre-indexada pode ser copiada ou importada de base no formato NSRL via parâmetro -importkff.
kffDb = d:/kff/kff.db

# Exclui do caso arquivos ignoráveis conforme base de hashes.
excludeKffIgnorable = true

# Ignora/descarta arquivos duplicados do restante do processamento e do caso.
# Pode agilizar bastante o processamento de casos com backups. Necessita habilitar o cálculo de hash.
ignoreDuplicates = false

# Habilita ordenação dos caracteres de PDFs por posição.
# Necessário para indexar corretamente PDFs rotacionados, mas dobra o tempo de processamento de PDFs
sortPDFChars = false

# Exporta as propriedades de todos os arquivos para o arquivo "Lista de Arquivos.csv"
exportFileProps = true

# Processa a assinatura dos arquivos
processFileSignatures = true

# Expande containers, como arquivos compactados, caixas de e-mails, emails e documentos office.
# Os tipos de arquivo a expandir podem ser alterados no arquivo conf/CategoriesToExpand.txt
expandContainers = true

# Indexa o conteúdo dos arquivos. Caso desabilitado, indexa apenas as propriedades.
indexFileContents = true

# Define caracteres adicionais a serem indexados além de letras e números, isto é, deixam de ser separadores
# Utilize espaço para separar os caracteres
extraCharsToIndex =

# Adiciona e processa áreas não alocadas de imagens via sleuthkit
addUnallocated = false

# Indexa o espaço não alocado adicionado
indexUnallocated = true

# Habilita ou não o carving. Opções avançadas de carving, como tipos de arquivo a processar e a recuperar,
# devem ser configuradas no arquivo conf/CarvingConfig.txt
enableCarving = true

# Tamanho em bytes de divisão do espaço não alocado
unallocatedFragSize = 104857600


########################################################################
# Configurações da Pesquisa
########################################################################

# Utiliza GraphicsMagick ao invés de ImageMagick para renderizar imagens não suportadas pela JVM.
useGM = true

# Timeout(s) para aguardar conversão externa entre formatos de imagem via IM/GM.
imgConvTimeout = 5

# Número de threads de geração de miniaturas na galeria. "default" utiliza o número de processadores.
# Diminua o valor caso haja problemas de consumo de memória.
galleryThreads = default


########################################################################
# Configurações OCR
########################################################################

# Os resultados dependem da qualidade e resolução das imagens e do tamanho e tipo das fontes utilizadas
# Opcionalmente, utilize o parâmetro -ocr "nome_bookmark" para restringir o OCR a determinado bookmark (pode ser utilizado várias vezes)

# DESABILITADO POR PADRÃO, POIS PODE DEMORAR VÁRIOS SEGUNDOS POR DIGITALIZAÇÃO
enableOCR = false

# Caminho relativo ou completo para OCR Tesseract
# Comente caso já esteja no PATH
TesseractPath = tools/tesseract

# Idioma do dicionário a ser utilizado pelo OCR. Estão incluídos português(por) e inglês(eng). Podem ser utilizados em conjunto: por+eng
OCRLanguage = por

# Modo de análise de layout do Tesseract, por ex: 1 - com OSD (orientation & script detection), 3 - sem OSD (padrão Tesseract)
pageSegMode = 1

# Tamanho mínimo em bytes de arquivos para aplicar OCR
minFileSize2OCR = 10000

# Tamanho máximo em bytes de arquivos para aplicar OCR
maxFileSize2OCR = 100000000

# Resolução em dpi para conversão de PDF's para imagem. Aumente caso a fonte das digitalizações seja pequena
pdfToImgResolution = 200

# Máximo de caracteres de texto que um PDF pode conter para ser aplicado OCR
maxPDFTextSize2OCR = 100
